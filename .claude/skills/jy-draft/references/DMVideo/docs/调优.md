"""
视频素材匹配引擎
功能：根据文案短句智能匹配视频素材片段
"""

from dataclasses import dataclass
from typing import List, Set, Optional, Tuple, Dict
import random
import json
from collections import defaultdict


@dataclass
class RecordInfo:
    """素材信息"""
    record_id: str
    keywords: Set[str]
    used_count: int
    video_duration: int  # 毫秒


@dataclass
class ShortSentence:
    """短句文案"""
    sentence_id: int
    text: str
    begin_time: int  # 微秒
    duration: int  # 微秒
    end_time: int  # 微秒
    keywords: Set[str]
    record_ids: List[str]
    record_infos: List[RecordInfo]


@dataclass
class RecordSegment:
    """素材片段"""
    record_id: str
    start_time: int  # 微秒
    end_time: int  # 微秒


class MatchingConfig:
    """匹配配置参数"""
    def __init__(self):
        # 权重配置
        self.weight_duration = 0.2  # 时长匹配权重
        self.weight_keyword = 0.6   # 关键词匹配权重
        self.weight_no_repeat = 0.8  # 不重复使用权重
        self.weight_usage = 0.8      # 使用次数权重
        self.weight_merge = 0.5      # 合并短句权重
        self.weight_random_cut = 0.3  # 随机截取权重
        
        # 阈值配置
        self.merge_threshold = 800000  # 合并短句的时长阈值（微秒）
        self.duration_ratio_excellent = 0.8  # 优秀时长利用率
        self.duration_ratio_good = 0.5       # 良好时长利用率
        self.duration_ratio_fair = 0.3       # 一般时长利用率
        
        # 惩罚系数
        self.penalty_used_record = -1.0  # 已使用素材的惩罚
        
        # 随机种子（可选，用于结果可复现）
        self.random_seed = None


class VideoMatcher:
    """视频匹配器"""
    
    def __init__(self, config: MatchingConfig = None):
        self.config = config or MatchingConfig()
        self.used_records: Set[str] = set()  # 已使用的素材ID
        self.record_usage_count: Dict[str, int] = defaultdict(int)  # 素材使用计数
        
        if self.config.random_seed is not None:
            random.seed(self.config.random_seed)

    def calculate_score(
        self, 
        sentence: ShortSentence, 
        record: RecordInfo,
        context: Optional[Dict] = None
    ) -> float:
        """
        计算素材与文案的匹配分数
        
        Args:
            sentence: 短句文案
            record: 素材信息
            context: 上下文信息（可选）
            
        Returns:
            匹配分数
        """
        score = 0.0
        
        # 1. 时长匹配度 (权重 0.2)
        duration_score = self._calculate_duration_score(sentence, record)
        score += duration_score * self.config.weight_duration
        
        # 2. 关键词匹配度 (权重 0.6)
        keyword_score = self._calculate_keyword_score(sentence, record)
        score += keyword_score * self.config.weight_keyword
        
        # 3. 素材未被使用 (权重 0.8)
        if record.record_id not in self.used_records:
            score += self.config.weight_no_repeat
        else:
            score += self.config.penalty_used_record  # 严重惩罚
        
        # 4. 使用次数惩罚 (权重 0.8)
        usage_score = self._calculate_usage_score(record)
        score += usage_score * self.config.weight_usage
        
        # 5. 上下文相关性（如果提供）
        if context:
            context_score = self._calculate_context_score(sentence, record, context)
            score += context_score * 0.4
        
        return score

    def _calculate_duration_score(
        self, 
        sentence: ShortSentence,
        record: RecordInfo
    ) -> float:
        """计算时长匹配度"""
        sentence_duration_ms = sentence.duration / 1000  # 转换为毫秒
        record_duration_ms = record.video_duration
        
        # 素材时长必须大于等于文案时长
        if record_duration_ms < sentence_duration_ms:
            return 0.0
        
        # 计算利用率
        ratio = sentence_duration_ms / record_duration_ms
        
        # 根据利用率分级评分
        if ratio >= self.config.duration_ratio_excellent:
            return 1.0
        elif ratio >= self.config.duration_ratio_good:
            return 0.7
        elif ratio >= self.config.duration_ratio_fair:
            return 0.4
        else:
            return 0.2

    def _calculate_keyword_score(
        self, 
        sentence: ShortSentence,
        record: RecordInfo
    ) -> float:
        """计算关键词匹配度"""
        if not sentence.keywords or not record.keywords:
            return 0.0
        
        # 计算交集
        common_keywords = sentence.keywords & record.keywords
        
        if not common_keywords:
            return 0.0
        
        # Jaccard相似度
        union_keywords = sentence.keywords | record.keywords
        jaccard = len(common_keywords) / len(union_keywords)
        
        # 额外奖励：匹配关键词数量
        match_bonus = min(len(common_keywords) * 0.1, 0.3)
        
        return min(jaccard + match_bonus, 1.0)

    def _calculate_usage_score(self, record: RecordInfo) -> float:
        """计算使用次数得分"""
        # 获取实际使用次数（包括本次匹配过程中的使用）
        actual_count = record.used_count + self.record_usage_count.get(record.record_id, 0)
        
        usage_map = {
            0: 1.0,
            1: 0.8,
            2: 0.6,
            3: 0.4,
            4: 0.2,
            5: 0.1
        }
        return usage_map.get(actual_count, 0.0)

    def _calculate_context_score(
        self, 
        sentence: ShortSentence,
        record: RecordInfo,
        context: Dict
    ) -> float:
        """计算上下文相关性得分"""
        score = 0.0
        
        # 检查前后句的关键词连贯性
        if 'prev_keywords' in context:
            prev_keywords = context['prev_keywords']
            if prev_keywords & record.keywords:
                score += 0.3
        
        if 'next_keywords' in context:
            next_keywords = context['next_keywords']
            if next_keywords & record.keywords:
                score += 0.3
        
        return min(score, 1.0)

    def should_merge_sentences(
        self, 
        sentence: ShortSentence,
        next_sentence: ShortSentence
    ) -> bool:
        """判断是否应该合并连续短句"""
        total_duration = sentence.duration + next_sentence.duration
        return total_duration <= self.config.merge_threshold

    def calculate_merge_score(
        self, 
        sentence: ShortSentence,
        next_sentence: ShortSentence,
        record: RecordInfo
    ) -> float:
        """计算合并短句的匹配度"""
        # 检查是否有共同关键词
        common_keywords = (
            sentence.keywords & 
            next_sentence.keywords & 
            record.keywords
        )
        
        if common_keywords:
            # 共同关键词越多，分数越高
            return min(len(common_keywords) * 0.3, 1.0)
        
        # 检查素材时长是否满足两个短句总时长
        total_duration_ms = (sentence.duration + next_sentence.duration) / 1000
        if record.video_duration >= total_duration_ms:
            # 根据时长利用率评分
            ratio = total_duration_ms / record.video_duration
            if ratio >= 0.8:
                return 0.9
            elif ratio >= 0.5:
                return 0.7
            else:
                return 0.5
        
        return 0.0


class VideoMatchingEngine:
    """视频匹配引擎"""
    
    def __init__(self, config: MatchingConfig = None):
        self.config = config or MatchingConfig()
        self.matcher = VideoMatcher(self.config)

    def match_videos(self, short_sentences: List[dict]) -> List[dict]:
        """
        主匹配函数
        
        Args:
            short_sentences: 短句列表
            
        Returns:
            匹配结果列表
        """
        # 解析输入数据
        sentences = self._parse_input(short_sentences)
        
        # 排序：按sentence_id排序
        sentences.sort(key=lambda x: x.sentence_id)
        
        results = []
        i = 0
        
        while i < len(sentences):
            sentence = sentences[i]
            next_sentence = sentences[i + 1] if i + 1 < len(sentences) else None
            
            # 构建上下文
            context = self._build_context(sentences, i)
            
            # 检查是否需要合并处理
            if (next_sentence and 
                self.matcher.should_merge_sentences(sentence, next_sentence)):
                
                # 尝试合并匹配
                result = self._match_merged_sentences(
                    sentence, 
                    next_sentence, 
                    context
                )
                
                if result:
                    results.append(result)
                    i += 2  # 跳过下一个句子
                else:
                    # 合并失败，单独匹配
                    result = self._match_single_sentence(sentence, context)
                    results.append(result)
                    i += 1
            else:
                result = self._match_single_sentence(sentence, context)
                results.append(result)
                i += 1
        
        return results

    def _build_context(
        self, 
        sentences: List[ShortSentence], 
        current_index: int
    ) -> Dict:
        """构建上下文信息"""
        context = {}
        
        if current_index > 0:
            context['prev_keywords'] = sentences[current_index - 1].keywords
        
        if current_index < len(sentences) - 1:
            context['next_keywords'] = sentences[current_index + 1].keywords
        
        return context

    def _match_single_sentence(
        self, 
        sentence: ShortSentence,
        context: Optional[Dict] = None
    ) -> dict:
        """为单个短句匹配素材"""
        # 计算所有候选素材的得分
        candidates = []
        
        for record_id in sentence.record_ids:
            record = next(
                (r for r in sentence.record_infos if r.record_id == record_id), 
                None
            )
            if not record:
                continue
            
            score = self.matcher.calculate_score(sentence, record, context)
            candidates.append((score, record))
        
        # 按得分排序
        candidates.sort(key=lambda x: x[0], reverse=True)
        
        # 生成素材片段
        segments = self._generate_segments(sentence, candidates)
        
        # 标记已使用的素材
        for seg in segments:
            self.matcher.used_records.add(seg['record_id'])
            self.matcher.record_usage_count[seg['record_id']] += 1
        
        return {
            'text': sentence.text,
            'begin_time': sentence.begin_time,
            'duration': sentence.duration,
            'keywords': ','.join(sorted(sentence.keywords)),
            'record_ids': [seg['record_id'] for seg in segments],
            'record_segments': segments,
            'sentence_id': sentence.sentence_id
        }

    def _generate_segments(
        self, 
        sentence: ShortSentence,
        candidates: List[Tuple[float, RecordInfo]]
    ) -> List[dict]:
        """生成素材片段列表"""
        segments = []
        remaining_duration = sentence.duration  # 微秒
        
        for score, record in candidates:
            if remaining_duration <= 0:
                break
            
            # 跳过已使用的素材
            if record.record_id in self.matcher.used_records:
                continue
            
            record_duration_us = record.video_duration * 1000  # 转换为微秒
            
            # 计算使用时长
            use_duration = min(remaining_duration, record_duration_us)
            
            # 随机选择起始点
            max_start = record_duration_us - use_duration
            if max_start > 0:
                # 使用正态分布使中间位置更可能被选中
                mean = max_start / 2
                std = max_start / 4
                start_time = int(random.gauss(mean, std))
                start_time = max(0, min(start_time, max_start))
            else:
                start_time = 0
            
            end_time = start_time + use_duration
            
            segments.append({
                'record_id': record.record_id,
                'start_time': start_time,
                'end_time': end_time
            })
            
            remaining_duration -= use_duration
        
        # 如果还有剩余时长，尝试使用已用过的素材（降级策略）
        if remaining_duration > 0 and not segments:
            # 使用得分最高的素材，即使已被使用
            if candidates:
                best_record = candidates[0][1]
                record_duration_us = best_record.video_duration * 1000
                use_duration = min(remaining_duration, record_duration_us)
                
                segments.append({
                    'record_id': best_record.record_id,
                    'start_time': 0,
                    'end_time': use_duration
                })
        
        return segments

    def _match_merged_sentences(
        self, 
        sentence1: ShortSentence,
        sentence2: ShortSentence,
        context: Optional[Dict] = None
    ) -> Optional[dict]:
        """为合并的短句匹配素材"""
        # 合并关键词
        merged_keywords = sentence1.keywords | sentence2.keywords
        total_duration = sentence1.duration + sentence2.duration
        
        # 查找共同的素材
        common_record_ids = set(sentence1.record_ids) & set(sentence2.record_ids)
        
        # 收集所有可能的素材
        all_records = {}
        for record in sentence1.record_infos + sentence2.record_infos:
            if record.record_id not in all_records:
                all_records[record.record_id] = record
        
        # 评分候选素材
        candidates = []
        for record_id in common_record_ids:
            record = all_records.get(record_id)
            if not record or record.record_id in self.matcher.used_records:
                continue
            
            # 计算合并得分
            merge_score = self.matcher.calculate_merge_score(
                sentence1, 
                sentence2, 
                record
            )
            
            # 计算基础得分
            base_score = (
                self.matcher.calculate_score(sentence1, record, context) +
                self.matcher.calculate_score(sentence2, record, context)
            ) / 2
            
            total_score = (
                merge_score * self.config.weight_merge + 
                base_score * (1 - self.config.weight_merge)
            )
            
            candidates.append((total_score, record))
        
        # 如果没有共同素材，尝试使用单个素材覆盖两个短句
        if not candidates:
            for record_id, record in all_records.items():
                if record.record_id in self.matcher.used_records:
                    continue
                
                record_duration_us = record.video_duration * 1000
                if record_duration_us >= total_duration:
                    # 计算得分
                    score = (
                        self.matcher.calculate_score(sentence1, record, context) +
                        self.matcher.calculate_score(sentence2, record, context)
                    ) / 2
                    candidates.append((score, record))
        
        if not candidates:
            return None  # 无法合并，返回None
        
        # 选择最佳素材
        candidates.sort(key=lambda x: x[0], reverse=True)
        best_record = candidates[0][1]
        
        # 生成片段
        record_duration_us = best_record.video_duration * 1000
        max_start = max(0, record_duration_us - total_duration)
        
        if max_start > 0:
            mean = max_start / 2
            std = max_start / 4
            start_time = int(random.gauss(mean, std))
            start_time = max(0, min(start_time, max_start))
        else:
            start_time = 0
        
        end_time = start_time + total_duration
        
        segment = {
            'record_id': best_record.record_id,
            'start_time': start_time,
            'end_time': end_time
        }
        
        # 标记已使用
        self.matcher.used_records.add(best_record.record_id)
        self.matcher.record_usage_count[best_record.record_id] += 1
        
        return {
            'text': f"{sentence1.text} {sentence2.text}",
            'begin_time': sentence1.begin_time,
            'duration': total_duration,
            'keywords': ','.join(sorted(merged_keywords)),
            'record_ids': [best_record.record_id],
            'record_segments': [segment],
            'sentence_id': sentence1.sentence_id
        }

    def _parse_input(self, short_sentences: List[dict]) -> List[ShortSentence]:
        """解析输入数据"""
        sentences = []
        
        for item in short_sentences:
            # 解析素材信息
            record_infos = []
            for i, info in enumerate(item['record_infos']):
                record_id = item['record_ids'][i]
                keywords = set(k.strip() for k in info['keywords'].split(',') if k.strip())
                
                record_info = RecordInfo(
                    record_id=record_id,
                    keywords=keywords,
                    used_count=info['used_count'],
                    video_duration=info['video_duration']
                )
                record_infos.append(record_info)
            
            # 解析短句
            keywords = set(k.strip() for k in item['keywords'].split(',') if k.strip())
            
            sentence = ShortSentence(
                sentence_id=item['sentence_id'],
                text=item['text'],
                begin_time=item['begin_time'],
                duration=item['duration'],
                end_time=item['end_time'],
                keywords=keywords,
                record_ids=item['record_ids'],
                record_infos=record_infos
            )
            sentences.append(sentence)
        
        return sentences


class MatchingAnalyzer:
    """匹配结果分析器"""
    
    @staticmethod
    def analyze_results(results: List[dict]) -> Dict:
        """分析匹配结果"""
        total_sentences = len(results)
        total_segments = sum(len(r['record_segments']) for r in results)
        unique_records = len(set(
            seg['record_id'] 
            for r in results 
            for seg in r['record_segments']
        ))
        
        # 计算平均利用率
        utilization_rates = []
        for result in results:
            for seg in result['record_segments']:
                duration = seg['end_time'] - seg['start_time']
                utilization_rates.append(duration)
        
        avg_utilization = (
            sum(utilization_rates) / len(utilization_rates) 
            if utilization_rates else 0
        )
        
        return {
            'total_sentences': total_sentences,
            'total_segments': total_segments,
            'unique_records_used': unique_records,
            'avg_segment_duration_us': avg_utilization,
            'segments_per_sentence': total_segments / total_sentences if total_sentences > 0 else 0
        }

            for seg in result['record_segments']
        )
        utilization_rates.append(segments_duration / sentence_duration)
    
    avg_utilization = sum(utilization_rates) / len(utilization_rates)
    score += avg_utilization * 0.3
    
    return score
